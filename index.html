🌊 저주파 vs 고주파
저주파 = 천천히 변하는 부분 (배경) / 고주파 = 빠르게 변하는 부분 (엣지, 경계선)
⭐ Nyquist 정리
Stride(간격)가 Nyquist frequency보다 크면 → 앨리어싱(aliasing) 발생! 정보 왜곡!
⭐ 보간법 (Interpolation) 핵심
Nearest Neighbor
가장 가까운 픽셀 값 그대로 복사. 빠르지만 계단 현상
Bilinear
주변 4개 점 선형 보간. 더 부드러운 결과
Bicubic
주변 16개 점으로 3차 함수 보간. 가장 부드러움
OpenCV 상수
INTER_NEAREST
INTER_LINEAR
INTER_CUBIC

실습 1 · Zone Plate 생성 + Sampling — 스크린샷 코드
np.linspace + np.sin 방식. Experiment A(aliasing) / B(anti-aliasing) 비교.
import numpy as np
import cv2

# 1. Zone Plate image generation function
def generate_zone_plate(size=512):
    x = np.linspace(-1, 1, size)
    y = np.linspace(-1, 1, size)
    X, Y = np.meshgrid(x, y)
    R = np.sqrt(X**2 + Y**2)
    # Design frequency increases with distance from center
    zone_plate = np.sin(100 * np.pi * R**2)
    # Convert to uint8 scale 0~255 (for OpenCV processing)
    zone_plate = ((zone_plate + 1) / 2 * 255).astype(np.uint8)
    return zone_plate

# Generate original Zone Plate
img    = generate_zone_plate(512)
h, w   = img.shape

# 2. 4x Down-sampling setting (induce Nyquist condition violation)
stride = 4

# [Experiment A] Sampling without filtering (Aliasing occurs)
aliased_sub      = img[::stride, ::stride]
# Restore to original size with Nearest Neighbor (shows pixel distortion as is)
aliased_restored = cv2.resize(aliased_sub, (w, h),
                              interpolation=cv2.INTER_NEAREST)

# [Experiment B] Anti-aliasing (Sampling after applying Lowpass Filter)
# Remove high-frequency components from the periphery in advance
blurred_img           = cv2.GaussianBlur(img, (15, 15), 5)
anti_aliased_sub      = blurred_img[::stride, ::stride]
anti_aliased_restored = cv2.resize(anti_aliased_sub, (w, h),
                                   interpolation=cv2.INTER_NEAREST)
Stride가 Nyquist frequency보다 크면 정보 왜곡이 생긴다!
실습 2 · Nearest Neighbor 보간 — 스크린샷 코드 (8/10)
UPSAM_BEFORE(간격만 배치) / UPSAM_AFTER(블록 채우기) 비교. cat.jpg 사용.
import cv2
import numpy as np
import matplotlib.pyplot as plt

img     = cv2.imread('cat.jpg')
# OpenCV(BGR) → Matplotlib(RGB) 채널 변환
img_rgb = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)

H, W, C = img_rgb.shape
UP = 2

# 1. Before interpolation
UPSAM_BEFORE = np.zeros((H*UP, W*UP, C), dtype=np.uint8)
for i in range(H):
    for j in range(W):
        UPSAM_BEFORE[i*UP, j*UP, :] = img_rgb[i, j, :]

# 2. After interpolation (Nearest Neighbor)
UPSAM_AFTER = np.zeros((H*UP, W*UP, C), dtype=np.uint8)
for i in range(H):
    for j in range(W):
        UPSAM_AFTER[i*UP:(i+1)*UP, j*UP:(j+1)*UP, :] = img_rgb[i, j, :]

# Matplotlib을 이용한 결과 출력
plt.figure(figsize=(12, 6))

plt.subplot(1, 2, 1)
plt.title('Before interpolation')
plt.imshow(UPSAM_BEFORE)
plt.axis('off')  # 축 눈금 숨김

plt.subplot(1, 2, 2)
plt.title('After interpolation')
plt.imshow(UPSAM_AFTER)
plt.axis('off')

plt.tight_layout()
plt.show()
실습 3 · Bilinear 보간 — 스크린샷 코드 (9/10)
image_lerp1 배열로 가로→세로 2단계 선형 보간. cat.jpg, BGR→RGB 변환 포함.
import cv2
import numpy as np
import matplotlib.pyplot as plt

plt.figure(figsize=(16, 12))

# 1. Load image and convert BGR to RGB for Matplotlib
image_bgr = cv2.imread('cat.jpg')
image     = cv2.cvtColor(image_bgr, cv2.COLOR_BGR2RGB)
h, w, c   = image.shape
scaling   = 2

# ==================== Step 1-1: Horizontal Spacing ====================
image_lerp1 = np.zeros((h, w*scaling, c), dtype=np.uint8)

# Modify j range to w (Prevent IndexError)
for i in range(h):
    for j in range(w):
        image_lerp1[i, 2*j, :] = image[i, j, :]

# Output Step 1-1
plt.subplot(2, 2, 1)
plt.title('Step 1-1: Horizontal Spacing')
plt.imshow(image_lerp1)
plt.axis('off')

# ==================== Step 1-2: Horizontal Interpolation ====================
for i in range(h):
    for j in range(w-1):
        # Calculate as int to prevent overflow, then restore to uint8
        val1 = image_lerp1[i, 2*j,   :].astype(int)
        val2 = image_lerp1[i, 2*j+2, :].astype(int)
        image_lerp1[i, 2*j+1, :] = ((val1 + val2) / 2).astype(np.uint8)

# Output Step 1-2
plt.subplot(2, 2, 2)
plt.title('Step 1-2: Horizontal Interpolation')
plt.imshow(image_lerp1)
plt.axis('off')

# ==================== Step 2-1: Vertical Spacing ====================
interpolated_image = np.zeros((h*scaling, w*scaling, c), dtype=np.uint8)

for i in range(h):
    for j in range(w*scaling):
        interpolated_image[2*i, j, :] = image_lerp1[i, j, :]

# Output Step 2-1
plt.subplot(2, 2, 3)
plt.title('Step 2-1: Vertical Spacing')
plt.imshow(interpolated_image)
plt.axis('off')

# ==================== Step 2-2: Vertical Interpolation ====================
for i in range(h-1):
    for j in range(w*scaling):
        # Calculate as int to prevent overflow, then restore to uint8
        val1 = interpolated_image[2*i,   j, :].astype(int)
        val2 = interpolated_image[2*i+2, j, :].astype(int)
        interpolated_image[2*i+1, j, :] = ((val1 + val2) / 2).astype(np.uint8)

# Output Step 2-2
plt.subplot(2, 2, 4)
plt.title('Step 2-2: Vertical Interpolation')
plt.imshow(interpolated_image)
plt.axis('off')

# Add main title for the entire figure
plt.suptitle('Bilinear Interpolation Process', fontsize=16)
plt.tight_layout()
plt.show()

import numpy as np
import cv2

# Zone Plate 생성
size = 512
x = np.arange(size) - size // 2
y = np.arange(size) - size // 2
X, Y = np.meshgrid(x, y)
r2   = X**2 + Y**2
zone = ((np.cos(r2 / 500.0) + 1) / 2 * 255).astype(np.uint8)
P, Q = zone.shape

# stride 2, 4, 8로 각각 서브샘플링 → 복원

    sampled  = zone[::stride, ::stride]
    restored = cv2.resize(sampled, (Q, P),
                          interpolation=cv2.INTER_NEAREST)
    cv2.imwrite(f'zone_stride{stride}.jpg', restored)

과제 3 · 수동 Bilinear vs cv2.INTER_LINEAR 비교
제출: 파이썬 1개 + 이미지 3개 (manual / opencv / diff)
import numpy as np
import cv2

img = cv2.imread('pandas.jpg', cv2.IMREAD_GRAYSCALE)
h, w = img.shape
new_h, new_w = h * 2, w * 2

# ── 수동 Bilinear 보간 ──────────────────────────
result_manual = np.zeros((new_h, new_w), dtype=np.float64)

for ny in range(new_h):
    for nx in range(new_w):
        # 원본 좌표로 역매핑
        ox = nx * (w - 1) / (new_w - 1)
        oy = ny * (h - 1) / (new_h - 1)

        x0, y0 = int(ox), int(oy)
        x1 = min(x0 + 1, w - 1)
        y1 = min(y0 + 1, h - 1)
        tx, ty = ox - x0, oy - y0

        # 가로 보간 2번 → 세로 보간 1번
        r1 = img[y0, x0] * (1 - tx) + img[y0, x1] * tx
        r2 = img[y1, x0] * (1 - tx) + img[y1, x1] * tx
        result_manual[ny, nx] = r1 * (1 - ty) + r2 * ty

result_manual = result_manual.astype(np.uint8)

# ── OpenCV Bilinear ────────────────────────────
result_cv2 = cv2.resize(img, (new_w, new_h),
                        interpolation=cv2.INTER_LINEAR)

# ── 두 결과의 차이 이미지 ──────────────────────
diff = cv2.absdiff(result_manual, result_cv2)

cv2.imwrite('bilinear_manual.jpg', result_manual)
cv2.imwrite('bilinear_cv2.jpg',    result_cv2)
cv2.imwrite('bilinear_diff.jpg',   diff)

과제 · 히스토그램 평활화 직접 구현
제출: 파이썬 1개 + 평활화 전/후 히스토그램 이미지 + 결과 이미지
공식: h(v) = round( (cdf(v) - cdf_min) / (cdf_max - cdf_min) × (L-1) )
import cv2
import numpy as np
import matplotlib.pyplot as plt

image = cv2.imread('bean.tif', 0)   # 흑백으로 읽기
L = 256

# ── Step 1: 히스토그램 계산 ─────────────────────
hist, _ = np.histogram(image.ravel(), L, [0, L])

# ── Step 2: CDF (누적합) 계산 ───────────────────
cdf = hist.cumsum()

# ── Step 3: cdf_min (0 제외한 최솟값) ───────────
cdf_min = cdf[cdf > 0].min()
cdf_max = cdf.max()    # = 전체 픽셀 수

# ── Step 4: 공식 적용 → LUT 생성 ────────────────
lut = np.round(
    (cdf - cdf_min) / (cdf_max - cdf_min) * (L - 1)
).astype(np.uint8)

# ── Step 5: LUT를 이미지에 적용 ─────────────────
result = lut[image]

# ── Step 6: 히스토그램 전/후 비교 저장 ──────────
fig = plt.figure()
ax0 = fig.add_subplot(211)
ax0.hist(image.ravel(),  L, [0, L])
ax0.set_title('Before HE')
ax1 = fig.add_subplot(212)
ax1.hist(result.ravel(), L, [0, L])
ax1.set_title('After HE')

Week 6 · Frequency Domain Filtering ⭐ 가장 중요!
핵심 요약
DFT, LPF/HPF, Butterworth, Gaussian, Laplacian, Homomorphic 필터링을 다룹니다.
⭐ 주파수 도메인 필터링 전체 절차
1
Zero-padding: 이미지 M×N → 2M×2N으로 확장
2
Centering: (-1)^(x+y) 곱해서 저주파를 중앙으로 이동
3
DFT: np.fft.fft2()로 주파수 도메인 변환
4
Filtering: G(u,v) = F(u,v) × H(u,v)
5
Inverse DFT: np.fft.ifft2() → de-centering → padding 제거
⭐ 필터 공식 — 암기 필수
Butterworth LPF (order n)
H(u,v) = 1 / (1 + [D(u,v)/D0]^(2n))
Gaussian LPF
H(u,v) = exp(-D²(u,v) / 2D0²)
Laplacian
H(u,v) = -(u² + v²) 샤프닝: f(x,y) - g(x,y) / 블러링: f(x,y) + g(x,y)

실습 2 · BLPF 함수 + 필터링 — PDF 코드
import numpy as np

# ── BLPF 함수 정의 ─────────────────────────────
def BLPF(image, order, cutoff):
    M, N = image.shape
    H, D = np.zeros((M, N)), np.zeros((M, N))

    U0 = int(M / 2)
    V0 = int(N / 2)

    D0 = cutoff
    n  = order

    # D(u, v) 계산
    for u in range(M):
        for v in range(N):
            u2 = np.power(u, 2)
            v2 = np.power(v, 2)
            D[u, v] = np.sqrt(u2 + v2)

    # H(u, v) 계산
    for u in range(M):
        for v in range(N):
            u_ = np.abs(u - U0)
            v_ = np.abs(v - V0)
            H[u, v] = 1 / (1 + (D[np.abs(u_), np.abs(v_)] / D0) ** (2 * n))

    return H

# ── 메인 코드 ─────────────────────────────────
# dft2d, padded_image_new 는 실습 1에서 이어서 사용

plt.show()

dft2d_ = np.log(np.abs(G))
plt.imshow(dft2d_.real, cmap='gray'), plt.axis('off')
plt.show()

# Inverse FFT
idft2d = np.fft.ifft2(G)

실습 4 · GLPF 함수 — PDF 코드 (BLPF에서 함수만 교체)
import numpy as np

def GLPF(image, cutoff):
    M, N = image.shape
    H, D = np.zeros((M, N)), np.zeros((M, N))

    U0 = int(M / 2)
    V0 = int(N / 2)

    D0 = cutoff

    # D(u, v) 계산
    for u in range(M):
        for v in range(N):
            u2 = np.power(u, 2)
            v2 = np.power(v, 2)
            D[u, v] = np.sqrt(u2 + v2)

    # Gaussian LPF: H = exp(-D²/2D0²)
    for u in range(M):
        for v in range(N):
            u_ = np.abs(u - U0)
            v_ = np.abs(v - V0)
            H[u, v] = np.exp(- D[u_, v_]**2 / (2 * (D0**2)))

    return H

과제 1 · BLPF vs GLPF 비교 (D0=30)
제출: 코드 1개(분석 주석 포함) + 결과 이미지 2개
import cv2
import numpy as np
import matplotlib.pyplot as plt

# ── BLPF 함수 ─────────────────────────────────
def BLPF(image, order, cutoff):
    M, N = image.shape
    H, D = np.zeros((M, N)), np.zeros((M, N))
    U0, V0 = int(M / 2), int(N / 2)
    D0, n  = cutoff, order

    for u in range(M):
        for v in range(N):
            D[u, v] = np.sqrt(np.power(u, 2) + np.power(v, 2))

    for u in range(M):
        for v in range(N):
            u_ = np.abs(u - U0)
            v_ = np.abs(v - V0)
            H[u, v] = 1 / (1 + (D[u_, v_] / D0) ** (2 * n))

    return H

# ── GLPF 함수 ─────────────────────────────────
def GLPF(image, cutoff):
    M, N = image.shape
    H, D = np.zeros((M, N)), np.zeros((M, N))
    U0, V0 = int(M / 2), int(N / 2)
    D0     = cutoff

    for u in range(M):
        for v in range(N):
            D[u, v] = np.sqrt(np.power(u, 2) + np.power(v, 2))

    for u in range(M):
        for v in range(N):
            u_ = np.abs(u - U0)
            v_ = np.abs(v - V0)
            H[u, v] = np.exp(- D[u_, v_]**2 / (2 * (D0**2)))

    return H

# ── 공통 전처리 ───────────────────────────────
image = cv2.imread('sample.jpg', 0)
M, N  = image.shape
P, Q  = 2 * M, 2 * N

padded[:M, :N] = image

# Centering
padded_c = np.zeros((P, Q))
for x in range(P):
    for y in range(Q):
        padded_c[x, y] = padded[x, y] * ((-1) ** (x + y))

# DFT
dft2d = np.fft.fft2(padded_c)

# ── 필터 적용 및 역변환 공통 함수 ──────────────
def apply_filter_and_save(dft2d, H, filename, M, N, P, Q):
    G      = np.multiply(dft2d, H)
    idft2d = np.fft.ifft2(G)

    for x in range(P):
        for y in range(Q):
            idft2d[x, y] = idft2d[x, y] * ((-1) ** (x + y))

    result = idft2d[:M, :N].real
    min_v, max_v = result.min(), result.max()
    result = ((result - min_v) / (max_v - min_v) * 255).astype(np.uint8)
    cv2.imwrite(filename, result)
    return result

제출: 코드 1개 + 이미지 2개 (laplacian_filtered.jpg / sharpened_result.jpg)
import cv2
import numpy as np

def Laplacian(image):
    M, N = image.shape
    H = np.zeros((M, N))
    for u in range(M):
        for v in range(N):
            u2 = np.power(u - M/2, 2)
            v2 = np.power(v - N/2, 2)
            H[u, v] = -(u2 + v2)
    return H

# 전처리 (Zero-padding + Centering)

padded = np.zeros((P, Q))
padded[:M, :N] = image

padded_c = np.zeros((P, Q))
for x in range(P):
    for y in range(Q):
        padded_c[x, y] = padded[x, y] * ((-1) ** (x + y))

dft2d = np.fft.fft2(padded_c)

# Laplacian 필터 적용
H_lap = Laplacian(dft2d)
G     = np.multiply(dft2d, H_lap)
